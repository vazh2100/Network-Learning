```bash
ip link       # Отображение списка сетевых интерфейсов
ip address    # Отображение информацию об ip адресах сетевых интерфесов
ifconfig      # interface config - устаревшая команда для настройки сетевых интерфейсов
ip -4 address # Отображение только ip адресов версии 4
```

`inet 127.0.0.1/8` - ip адрес интерфейса

`sudo ip link set wlan0 up` - включение сетевого интерфейса на уровне ядра
`sudo ip link set wlan0 down` -выключение сетевого интерфейса на уровне ядра

`sudo nmcli device up wlan0` - включение сетевого интерфейса согласовано с менеджером соединений
`sudo nmcli device down wlan0` - выключение сетевого интерфейса согласовано с менеджером соединений
