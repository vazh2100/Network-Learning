```bash
ip link       # Отображение списка сетевых интерфейсов
ip address    # Отображение информацию об ip адресах сетевых интерфесов
ifconfig      # interface config - устаревшая команда для настройки сетевых интерфейсов
ip -4 address # Отображение только ip адресов версии 4
```

`inet 127.0.0.1/8` - ip адрес интерфейса
